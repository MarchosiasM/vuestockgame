const getters = {
  funds: state => state.funds,
  personalPortfolio: state => state.personalPortfolio,
  companies: state => state.companies,
  prices: state => state.prices,
  hydrateMyStocks: (state) => {
    const { personalPortfolio, prices } = state;
    return personalPortfolio.map(value => ({
      name: value[1].name,
      index: value[1].index,
      count: value[1].count,
      price: prices[value[1].name],
    }));
  },
  fullCompanyData: (state) => {
    const { companies, prices } = state;
    console.log(Array.from(companies));
    return Array.from(companies).map(value => ({
      name: value[1].name,
      index: value[1].index,
      count: value[1].count,
      price: prices[value[1].name],
    }));
  },
};
 
export default getters;
